calculator

<!DOCTYPE html>
<html lang="en">
<head>
  <style>
    input { width: 100px; }
    button { margin: 5px; }
  </style>
</head>
<body>

  <input type="number" id="a" placeholder="Num 1">
  <input type="number" id="b" placeholder="Num 2">
  <br>
  <button onclick="calc('+')">+</button>
  <button onclick="calc('-')">-</button>
  <button onclick="calc('*')">×</button>
  <button onclick="calc('/')">÷</button>
  <p id="result"></p>

  <script>
    function calc(op) {
      let a = parseFloat(document.getElementById('a').value);
      let b = parseFloat(document.getElementById('b').value);
      let result;

      if (op === '+') result = a + b;
      else if (op === '-') result = a - b;
      else if (op === '*') result = a * b;
      else if (op === '/') result = b !== 0 ? a / b : "Can't divide by 0";

      document.getElementById('result').innerText = 'Result: ' + result;
    }
  </script>

</body>
</html>



to-do list


<!DOCTYPE html>
<html lang="en">
<head>
  <style>
    .done { text-decoration: line-through; color: gray; }
  </style>
</head>
<body>

  <input type="text" id="task" placeholder="Enter task">
  <button onclick="addTask()">Add</button>
  <ul id="list"></ul>

  <script>
    function addTask() {
      let task = document.getElementById('task').value;
      if (!task) return;

      let li = document.createElement('li');
      li.innerHTML = `
        <span onclick="this.classList.toggle('done')">${task}</span>
        <button onclick="this.parentElement.remove()">Delete</button>
      `;
      document.getElementById('list').appendChild(li);
      document.getElementById('task').value = '';
    }
  </script>

</body>
</html>


timer


<!DOCTYPE html>
<html lang="en">
<body>

  <input type="number" id="mins" placeholder="Enter minutes">
  <button onclick="startTimer()">Start</button>
  <p id="display"></p>

  <script>
    let timer;

    function startTimer() {
      clearInterval(timer);
      let seconds = document.getElementById('mins').value * 60;

      timer = setInterval(() => {
        if (seconds <= 0) {
          clearInterval(timer);
          document.getElementById('display').innerText = "Time's up!";
          alert("Time's up!");
          return;
        }
        seconds--;
        let m = Math.floor(seconds / 60);
        let s = seconds % 60;
        document.getElementById('display').innerText = `${m}:${s < 10 ? '0' : ''}${s}`;
      }, 1000);
    }
  </script>

</body>
</html>


age calculator

<!DOCTYPE html>
<html lang="en">
<body>

  <input type="date" id="dob">
  <button onclick="calcAge()">Calculate</button>
  <p id="result"></p>

  <script>
    function calcAge() {
      let dob = new Date(document.getElementById('dob').value);
      let today = new Date();
      let age = today.getFullYear() - dob.getFullYear();
      if (today < new Date(today.getFullYear(), dob.getMonth(), dob.getDate())) age--;
      document.getElementById('result').innerText = 'Your age: ' + age;
    }
  </script>

</body>
</html>

react calculator

import { useState } from "react";

export default function App() {
  const [a, setA] = useState(0), [b, setB] = useState(0), [result, setResult] = useState("");

  const calc = (op) => {
    if (op === "+") setResult(a + b);
    else if (op === "-") setResult(a - b);
    else if (op === "*") setResult(a * b);
    else setResult(b !== 0 ? a / b : "Can't divide by 0");
  };

  return (
    <div>
      <input type="number" onChange={e => setA(Number(e.target.value))} placeholder="Num 1" />
      <input type="number" onChange={e => setB(Number(e.target.value))} placeholder="Num 2" />
      <br />
      <button onClick={() => calc("+")}>+</button>
      <button onClick={() => calc("-")}>-</button>
      <button onClick={() => calc("*")}>×</button>
      <button onClick={() => calc("/")}>÷</button>
      <p>Result: {result}</p>
    </div>
  );
}

  const [count, setCount] = useState(0);
  return (
    <div>
      <h2>{count}</h2>
      <button onClick={() => setCount(count + 1)}>+</button>
      <button onClick={() => setCount(count - 1)}>-</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  );
}

react temperature convertor

import { useState } from "react";

export default function App() {
  const [cel, setCel] = useState(""), [fah, setFah] = useState("");
  return (
    <div>
      <input placeholder="Celsius" value={cel} onChange={e => { setCel(e.target.value); setFah((e.target.value * 9/5 + 32).toFixed(1)); }} />
      <input placeholder="Fahrenheit" value={fah} onChange={e => { setFah(e.target.value); setCel(((e.target.value - 32) * 5/9).toFixed(1)); }} />
    </div>
  );
}
